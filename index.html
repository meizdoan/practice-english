(Phần database giữ nguyên như cũ) ... let currentTrans = {}; function nextTrans() { const topic = document.getElementById('topic-select').value; const level = document.getElementById('level-select').value; const filtered = translationDB.filter(q => q.topic === topic && q.level === level); if(filtered.length > 0) { currentTrans = filtered[Math.floor(Math.random() * filtered.length)]; document.getElementById('vietnamese-task').innerText = `"${currentTrans.vi}"`; document.getElementById('translation-feedback').classList.add('hidden'); document.getElementById('translation-input').value = ""; document.getElementById('translation-input').focus(); // Tự động focus để gõ tiếp } } // Hàm xử lý kiểm tra kết quả function processCheck() { const inputEl = document.getElementById('translation-input'); const userAns = inputEl.value.trim().toLowerCase().replace(/[.,?!]/g, ""); const feedback = document.getElementById('translation-feedback'); if(!userAns) return; // So khớp với danh sách đáp án const isCorrect = currentTrans.en.some(ans => ans.toLowerCase().replace(/[.,?!]/g, "") === userAns ); feedback.classList.remove('hidden'); if (isCorrect) { feedback.innerHTML = "✅ Chính xác! Đang chuyển câu..."; feedback.style.backgroundColor = "#f0fdf4"; // Màu nền xanh nhẹ // Đợi 1.2 giây rồi tự nhảy sang câu mới setTimeout(() => { nextTrans(); }, 1200); } else { feedback.innerHTML = `❌ Chưa đúng.
Đáp án: ${currentTrans.en[0]}`; feedback.style.backgroundColor = "#fef2f2"; // Màu nền đỏ nhẹ } } // Sự kiện nhấn Enter trong ô dịch document.getElementById('translation-input').addEventListener('keydown', function(e) { if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); // Chặn xuống dòng processCheck(); } }); // Gán hàm vào nút "Kiểm tra" (đảm bảo ở phần HTML nút này có onclick="processCheck()")
